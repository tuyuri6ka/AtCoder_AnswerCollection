import java.util.*;


public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        String S = sc.next();
        String[] leetBefore = {"A","E","G","I","O","S","Z"};
        String[] leetAfter =  {"4","3","6","1","0","5","2"};
        
        for(int i= 0; i < 7; i++){
            S = S.replace(leetBefore[i],leetAfter[i]);
        }
        System.out.println(S);

    }
}
