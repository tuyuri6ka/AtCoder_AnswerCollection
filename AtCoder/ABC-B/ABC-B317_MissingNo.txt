import java.util.*;

public class Main {
	public static void main(String[] args) throws Exception {
		Scanner sc = new Scanner(System.in);

		int N = sc.nextInt();
		int[] A = new int[N];

		for (int i = 0; i < N; i++) {
			A[i] = sc.nextInt();
		}

		Arrays.sort(A);

		int cnt = A[0];
		int ans = 0;
		for (int i = 0; i < N; i++) {
			if (cnt != A[i]) {
				ans = A[i];
				break;
			}
			cnt++;
		}

		System.out.println(cnt);
	}
}
